SQL Tasks for Day 2 of Internship

1. Retrieve the Student CGPA, Company ID (CID), Job Role, and Package of all the Students.

SELECT CGPA,CID,Job_role,Package FROM Students

2. Retrieve the Lecturer ID (LID), Name, and Designation of all the Lecturers.

SELECT LID,First_name,Designation FROM Lecturers

3. Retrieve the Company ID (CID), Company Name, and the Students Placed in all the Companies.

SELECT CID,Company_name,Students_placed FROM Companies

4. Retrieve the Students Branch, College, Semester, and Section of all the Students.

SELECT Branch,College,Semester,Section FROM Students

5. Retrieve the Test 3 Marks of all the Students.

SELECT Test3 FROM Marks

6. Retrieve the Subject Name and Semester of all the Subjects.

SELECT Subject_name,Semester FROM Subjects

7. Retrieve the List of all the Branch IDs (BID).

SELECT BID FROM Branches

8. Retrieve the USN, Name, College, Section, and Branch of all the Students.

SELECT Usn,First_name,College,Section,Branch FROM Students

9. Retrieve the USN, Test1, Test2, and Test3 Marks of all the Students.

SELECT Usn,Test1,Test2,Test3 FROM Marks

10. Retrieve the Designation of all the Lecturers.

SELECT Designation FROM Lecturers

11. Retrieve the Name and Gender of all the Students.

SELECT First_name,Gender FROM Students

12. Retrieve the Section of all the Students.

SELECT Section FROM Students

13. Retrieve the Date of Joining (DOJ) of all the Students.

SELECT DOJ FROM Students

14. Retrieve the List of all the Subjects names.

SELECT Subject_name FROM Subjects

15. Retrieve the Lecturer Name, College, and Branch of all the Lecturers.

SELECT First_name,College,Branch FROM Lecturers
