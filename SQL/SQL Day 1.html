7. List of Lecturers with 'Professor' in their Designation

SELECT First_name, Designation FROM Lecturers WHERE Designation LIKE '%Professor%'

8. Retrieve the List of all Students

9. Retrieve the First Name of all Lecturers

SELECT First_name FROM Lecturers

10. Find Students in Branches Starting with 'CSE'

SELECT Usn, Branch FROM Students WHERE Branch LIKE 'CSE%'

11. Retrieve Subjects Related to 'Data'

SELECT Sid, Subject_name FROM Subjects WHERE Subject_name LIKE '%Data%'

12. Retrieve the List of all Subjects

SELECT * FROM Subjects

13. Find Students Whose First Names Start with 'S'

SELECT Usn, First_name FROM Students WHERE First_name LIKE 'S%'

14. Retrieve the List of all Company Names

SELECT Company_name FROM Companies

15. List Companies Located in Bengaluru

SELECT Company_name FROM Companies WHERE Company_name LIKE '%Bengaluru%'
